Это можно осуществить с помощью временной переменной такого же типа, в нашем случае это будет переменная temp.
```cpp
#include <iostream>
using namescape std;

int main()
{
int a = 5, b = 10, temp;

cout << "До замены:" << endl;
cout << "a = " << a << ", b = " << b << endl;

temp = a;
a = b;
b = temp;

cout << "\nПосле замены:" << endl;
cout << "a = " << a << ", b = " << b << endl;
}
```

Также это можно сделать с помощью арифметических операций, не используя лишние переменные
```cpp
#include <iostream>
using namescape std;

int main()
{
int a = 1, b = 2;

cout << "До замены:" << endl;
cout << "a = " << a << ", b = " << b << endl;

a = a + b // a = 1 + 2 = 3
b = a - b; // b = 3 - 2 = 1
a = a - b; // a = 3 - 1 = 2

cout << "\nПосле замены:" << endl;
cout << "a = " << a << ", b = " << b << endl;
}
```
